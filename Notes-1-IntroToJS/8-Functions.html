Functions 方法/函数

Define function and call it 定义方法，然后使用它

Example code: function 方法

// Define function 定义一个新方法
var drawFace = function(){
    var faceX = random(0, 400);
    var faceY = random(0, 400);
    var faceSize = random(50, 150);
    
    // face
    noStroke();
    fill(238, 255, 0);
    ellipse(faceX, faceY, faceSize, faceSize);
    // eyes
    fill(20, 18, 18);
    ellipse(faceX-faceSize*0.15, faceY-faceSize*0.2, faceSize*0.2, faceSize*0.2);
    ellipse(faceX+faceSize*0.25, faceY-faceSize*0.2, faceSize*0.2, faceSize*0.2);
    // mouth
    fill(224, 22, 22);
    ellipse(faceX+faceSize*0.05, faceY+faceSize*0.2, faceSize*0.4, faceSize*0.5);
};

// Use or say call the function 
// 使用或者叫"调用"我们定义的方法
drawFace();
drawFace();
drawFace();

By defining code as function, we make the code reusable. Great!

把一段代码定义成“方法”，我们就可以方便地复用（重复使用）这些代码了！牛～～

Function parameters 使用参数，更牛掰地使用方法

Example code: function parameters 方法使用参数

Use parameters to have more control on the function, by which we draw the winston at specified positions and sizes.

使用参数，来更好地控制我们的方法：让飞机往哪儿飞～～😄

其实是告诉方法：具体在哪儿画这些脸？需要画多大？

// Define a function 定义一个方法，是带参数的
var drawWinston = function(faceX, faceY, faceSize){
    // face
    noStroke();
    fill(238, 255, 0);
    ellipse(faceX, faceY, faceSize, faceSize);
    
    // eyes
    fill(20, 18, 18);
    ellipse(faceX-faceSize*0.15, faceY-faceSize*0.2, faceSize*0.2, faceSize*0.2);
    ellipse(faceX+faceSize*0.25, faceY-faceSize*0.2, faceSize*0.2, faceSize*0.2);
    
    // mouth
    fill(224, 22, 22);
    ellipse(faceX+faceSize*0.05, faceY+faceSize*0.2, faceSize*0.4, faceSize*0.5);
};


// Call function with parameters 调用方法，给出具体参数～
drawWinston(282, 287, 150);
drawWinston(100, 300, 80);
drawWinston(random(50, 350), random(100,350), random(100,200));

Function return values 函数返回值

Example code: calculator 计算器示例

var add = function(a, b){
    var result = a + b;
    return result;
};

var subtract = function(a, b){
    return a - b;
};

var multiply = function(a, b){
    return a * b;
};

var divide = function(a, b){
    return a / b;
};

Local and global variables 局部和全局变量

  • 局部变量：只在函数里面使用
  • 全局变量：在函数外面定义了，函数里外都可使用

Example code: grown height 分阶段长个儿，一共长了多高

var lifeHeight = 50;

...

var computeHeight = function(startAge, endAge, grownHeightPerYear) {
    // 这一阶段长高了多少
    var periodHeight = grownHeightPerYear * (endAge - startAge);
    // 叠加到实际身高上，现在有多高
    lifeHeight += periodHeight;
    // 这里，只返回这一阶段长高了多少
    return periodHeight;
};

...

text(computeHeight(0, 8, 9) + " cm", 240, 80);
text(computeHeight(8, 16, 8)  + " cm", 240, 120);
text(lifeHeight  + " cm", 240, 160);

We can find more ProcessingJS built-in functions in documentation. 更多直接可用（内建）的方法，可在文档中找到～

Example code: Pull back the car 拽住前进的汽车！

// global variabel: position of the car
// 全局变量：车的水平位置
var x = 11;

var draw = function() {
    background(252, 255, 214);

    // draw the car body 车身
    fill(255, 0, 115);
    rect(x, 200, 100, 20);
    rect(x + 15, 178, 70, 40);
    
    // draw the wheels 轱辘
    fill(77, 66, 66);
    ellipse(x + 25, 221, 24, 24);
    ellipse(x + 75, 221, 24, 24);
    
    // car move forward (right) 车向右前进
    x = x + 3;
};

mouseClicked = function() {
    // pull the car back (left) 把它拽回来～
    x = x - 50;
};
